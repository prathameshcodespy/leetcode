class Solution:
    def maxArea(self, height):
        l, r = 0, len(height) - 1
        ans = 0

        while l < r:
            h = min(height[l], height[r])
            ans = max(ans, (r - l) * h)

            if height[l] < height[r]:
                l += 1
            else:
                r -= 1

        return ans
