import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();
        
        while (t-- > 0) {
            int n = sc.nextInt();
            String s = sc.next();
            
            int openRound = 0, closeRound = 0;
            int openSquare = 0, closeSquare = 0;
            
            for (char c : s.toCharArray()) {
                if (c == '(') openRound++;
                else if (c == ')') closeRound++;
                else if (c == '[') openSquare++;
                else closeSquare++;
            }
            
            int roundFix = Math.abs(openRound - closeRound) / 2;
            int squareFix = Math.abs(openSquare - closeSquare) / 2;
            
            System.out.println(roundFix + squareFix);
        }
    }
}
