import sys
import threading

def solve():
    import sys
    input = sys.stdin.readline
    
    t = int(input())
    for _ in range(t):
        x, k = map(int, input().split())
        
        if k == 1:
            print(1)
            continue
        
        # highest power of 2 <= (k-1)
        p = 1 << ((k - 1).bit_length() - 1)
        
        print(2 * p - 1)

if __name__ == "__main__":
    solve()
